m_ur

## 
## Coefficients:
##       intercept
##          0.0103
## s.e.     0.0023
## 
## sigma^2 estimated as 0.004055:  log likelihood = 1007.92,  aic = -2013.84
x_t=c(1:755)
y_t=x_t*0.0103
time=ts(x_t,frequency=12,start=c(1946,2))
p_mmm=c(1:755)/755;
p_mmm[1]=mmm[1];
for (i in 2:755) {p_mmm[i]=p_mmm[i-1]+mmm[i]}
p_mmm_dt=p_mmm-y_t
plot(time,p_mmm)
lines(x_t,y_t,lty=2)
lines(x_t,p_mmm_dt,lty=2)
